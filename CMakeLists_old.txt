project(aruco_gate_detector)

  sensor_msgs

  roscpp

  #image_transport

)

                cv_bridge
                roscpp
                std_msgs

                geometry_msgs                   
)

target_link_libraries(aruco_gate_detector ${catkin_LIBRARIES} ${OpenCV_LIBS})
